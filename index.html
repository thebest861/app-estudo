function gerarQuestao(materia){ let pergunta, alternativas, correta, explicacao; function rand(max){ return Math.floor(Math.random()*max); } // ===================== MATEMÁTICA ===================== if(materia==="matematica"){ let a = rand(100); let b = rand(100); if(rand(2)==0){ let r=a+b; pergunta=`${a} + ${b} = ?`; alternativas=[r,r+1,r-1,r+2]; correta=0; explicacao=`${a} + ${b} = ${r}`; }else{ let r=a-b; pergunta=`${a} - ${b} = ?`; alternativas=[r,r+1,r-1,r+2]; correta=0; explicacao=`${a} - ${b} = ${r}`; } } // ===================== PORTUGUÊS ===================== if(materia==="portugues"){ let nomes=["João","Maria","Pedro","Ana","Lucas","Julia"]; let coisas=["fera","gênio","máquina","craque"]; let tipo = rand(3); // SENTIDO FIGURADO if(tipo==0){ let n = nomes[rand(nomes.length)]; let c = coisas[rand(coisas.length)]; pergunta=`Na frase "${n} é um ${c} estudando", o sentido é:`; alternativas=["Real","Figurado","Formal","Informal"]; correta=1; explicacao="Não é literal, quer dizer que ele é muito bom."; } // SUBSTANTIVO if(tipo==1){ let objetos=["Casa","Carro","Escola","Livro"]; let erradas=["Correr","Bonito","Rápido"]; let certo = objetos[rand(objetos.length)]; pergunta="Qual é um substantivo?"; alternativas=[certo,...erradas]; correta=0; explicacao="Substantivo é nome de coisa ou lugar."; } // CONCORDÂNCIA if(tipo==2){ let plural=["Os meninos felizes","As meninas bonitas"]; let errado=["Os menino feliz","As menina bonita"]; let certo = plural[rand(plural.length)]; pergunta="Qual frase está correta?"; alternativas=[certo,...errado]; correta=0; explicacao="Plural deve combinar com plural."; } } // ===================== CIÊNCIAS ===================== if(materia==="ciencias"){ let tipo = rand(3); // DIA E NOITE if(tipo==0){ pergunta="O movimento da Terra causa:"; alternativas=["Dia e noite","Chuva","Vento","Lua"]; correta=0; explicacao="A Terra gira e temos dia e noite."; } // LUA if(tipo==1){ pergunta="A Lua muda de forma porque:"; alternativas=["Fases da Lua","Vento","Chuva","Sol some"]; correta=0; explicacao="A Lua tem fases."; } // ÓPTICA if(tipo==2){ pergunta="Lupa serve para:"; alternativas=["Ampliar","Esfriar","Aquecer","Medir"]; correta=0; explicacao="A lupa aumenta o tamanho."; } } // ===================== HISTÓRIA ===================== if(materia==="historia"){ let tipo = rand(3); // NOMADISMO if(tipo==0){ pergunta="Os povos nômades:"; alternativas=["Mudam de lugar","Vivem em cidade","Plantam sempre","Não saem"]; correta=0; explicacao="Eles se moviam sempre."; } // SEDENTARISMO if(tipo==1){ pergunta="Sedentários são pessoas que:"; alternativas=["Ficam em um lugar","Viajam sempre","Só pescam","Só caçam"]; correta=0; explicacao="Eles vivem em um local fixo."; } // CIDADES if(tipo==2){ pergunta="As cidades surgiram porque:"; alternativas=["As pessoas se fixaram","Choveu muito","Fugiram","Não plantavam"]; correta=0; explicacao="As pessoas começaram a morar em um lugar."; } } // ===================== GEOGRAFIA ===================== if(materia==="geografia"){ let tipo = rand(3); // DESIGUALDADE if(tipo==0){ pergunta="Desigualdade territorial é:"; alternativas=["Diferença entre lugares","Igualdade","Clima","Solo"]; correta=0; explicacao="Alguns lugares são mais ricos."; } // POPULAÇÃO if(tipo==1){ pergunta="População é:"; alternativas=["Número de pessoas","Clima","Solo","Planta"]; correta=0; explicacao="População é quantidade de pessoas."; } // REGIÕES if(tipo==2){ pergunta="Região é:"; alternativas=["Parte de um lugar","Cidade inteira","País todo","Rua"]; correta=0; explicacao="Região é uma parte."; } } // 🔥 EMBARALHAR let op = alternativas.map((v,i)=>({txt:v,correta:i===correta})); op = embaralhar(op); let novaCorreta = op.findIndex(x=>x.correta); return {pergunta,op,correta:novaCorreta,explicacao}; }
